// استبدل كود الـ alert القديم بهذا الكود الاحترافي function showSuccessMessage() { const modal = document.createElement('div'); modal.innerHTML = `

شكراً لثقتك بمتجر Nice One To

سيتم التواصل معكم لتأكيد الطلب في أقرب وقت.

`; document.body.appendChild(modal); } // استدعاء الدالة عند الضغط على "تأكيد الطلب" document.querySelector('.btn-confirm').addEventListener('click', function(e) { e.preventDefault(); showSuccessMessage(); });
